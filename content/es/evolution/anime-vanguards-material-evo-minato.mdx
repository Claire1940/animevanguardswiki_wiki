  <Step num="1" title="Adquiere a Fastest Shinobi en el Banner de Invocación">

  <Step num="2" title="Farmea Tokens de Incursión para el Kunai de Teletransportación">

  <Step num="3" title="Completa Escenarios de Leyenda en la Aldea de la Arena para obtener Esencias Amarillas">

  </Step>
