<Steps>
  <Step num="1" title="Aquisição">

  </Step>
  <Step num="2" title="Nivelamento Base">
    Foque em elevar Kurumi ao seu nível base máximo usando Ouro e materiais de experiência. Isso aumenta significativamente os atributos base.
  </Step>
  <Step num="3" title="Otimização de Traços">

  </Step>
  <Step num="4" title="Evolução">

  </Step>
</Steps>
